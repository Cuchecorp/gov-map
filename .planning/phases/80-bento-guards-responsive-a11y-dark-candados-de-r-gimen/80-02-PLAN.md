`stripTsComments` (REUSAR VERBATIM de anti-insinuacion-guard.test.ts líneas 69-81 — NO re-implementar):
```typescript
function stripTsComments(content: string): string {
  let stripped = content.replace(/\/\*[\s\S]*?\*\//g, "");
  stripped = stripped.split("\n").map((line) => {
    const idx = line.search(/(?<!:)\/\//);   // NO cortar en :// (URLs)
    return idx >= 0 ? line.slice(0, idx) : line;
  }).join("\n");
  return stripped;
}
```
